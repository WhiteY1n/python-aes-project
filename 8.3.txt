#include <Arduino_FreeRTOS.h>
#include <timers.h>

// Chu kỳ ban đầu của auto-reload timer là 500 ms
#define mainAUTO_RELOAD_TIMER_PERIOD pdMS_TO_TICKS(500)

// Chu kỳ mới của auto-reload timer sau khi đổi là 1000 ms
#define mainAUTO_RELOAD_TIMER_PERIOD2 pdMS_TO_TICKS(1000)

// Handle dùng để quản lý software timer
TimerHandle_t xAutoReloadTimer;

// Biến lưu kết quả khởi động timer
BaseType_t xTimer2Started;

void setup() {
  // Khởi tạo giao tiếp Serial để hiển thị kết quả ra Serial Monitor
  Serial.begin(9600);

  // Tạo auto-reload timer
  // Timer này sẽ tự lặp lại sau mỗi chu kỳ
  xAutoReloadTimer = xTimerCreate(
    "AutoReload",                    // Tên timer
    mainAUTO_RELOAD_TIMER_PERIOD,    // Chu kỳ ban đầu là 500 ms
    pdTRUE,                          // pdTRUE nghĩa là timer tự nạp lại
    0,                               // Giá trị ID ban đầu của timer
    prvTimerCallback                 // Hàm callback khi timer hết thời gian
  );

  // Kiểm tra timer đã được tạo thành công hay chưa
  if (xAutoReloadTimer != NULL) {

    // Khởi động timer, thời gian chờ gửi lệnh vào queue là 0
    xTimer2Started = xTimerStart(xAutoReloadTimer, 0);

    // Nếu timer khởi động thành công thì bắt đầu bộ lập lịch FreeRTOS
    if (xTimer2Started == pdPASS) {
      vTaskStartScheduler();
    }
  }
}

void loop() {
  
}

static void prvTimerCallback(TimerHandle_t xTimer) {
  TickType_t xTimeNow;
  uint32_t ulExecutionCount;

  // Lấy ID hiện tại của timer
  // Ở đây ID được dùng như một biến đếm số lần timer đã hết chu kỳ
  ulExecutionCount = (uint32_t)pvTimerGetTimerID(xTimer);

  // Tăng số lần thực thi lên 1
  ulExecutionCount++;

  // Gán lại giá trị mới vào ID của timer
  vTimerSetTimerID(xTimer, (void*)ulExecutionCount);

  // Lấy số tick hiện tại của hệ thống
  xTimeNow = xTaskGetTickCount();

  // In ra thời điểm callback được thực thi
  Serial.print("Auto-reload timer callback executing ");
  Serial.println(xTimeNow / 31);

  // Sau khi timer chạy đủ 5 lần thì đổi chu kỳ từ 500 ms sang 1000 ms
  if (ulExecutionCount >= 5) {
    xTimerChangePeriod(
      xAutoReloadTimer,               // Timer cần thay đổi chu kỳ
      mainAUTO_RELOAD_TIMER_PERIOD2,  // Chu kỳ mới là 1000 ms
      0                               // Không block khi gửi lệnh đổi chu kỳ
    );
  }
}
